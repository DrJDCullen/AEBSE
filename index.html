function openPage(pageName,elmnt,color) {
var i, tabcontent, tablinks;

for (i = 0; i < tabcontent.length; i++) {
tabcontent[i].style.display = "none";
}
tablinks = document.getElementsByClassName("tablink");

}

elmnt.style.backgroundColor = color;
}


function KeyUpHappened(e)
{
if (!e) e=window.event;
var code;
if ((e.charCode) && (e.keyCode==0))
code = e.charCode
else
code = e.keyCode;
switch(code){
case 48:
location.reload(true);
break;
case 49:
document.getElementById("T1").click();
break;
case 50:
document.getElementById("T2").click();
break;
case 51:
document.getElementById("T3").click();
break;

document.getElementById("T4").click();
break;
case 53:

break;
default:
}
}

function reloadtime () {
$.getJSON('http://time.jsontest.com', function(data) {
var text = `<Font size=7>Date: ${data.date}<br><br>
Time: ${data.time} (UTC+0)<br><br><br></Font>
<Font size=1>Unix time: ${data.milliseconds_since_epoch}</font>`

});
timeout = setTimeout(reloadtime, 1000);
}
